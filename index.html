    // 根据传入的 节点（element） 和选择器（selector） 查询满足条件的节点元素
    zepto.matches = function (element, selector) {
      if (!selector || !element || element.nodeType !== 1) return false;

      // 当然这里写这么多是为了兼容性考虑
      var matchesSelector =
        element.matches ||
        element.webkitMatchesSelector ||
        element.mozMatchesSelector ||
        element.oMatchesSelector ||
        element.matchesSelector;

      // 如果你的环境支持这种查询那么就将查询到的元素返回
      if (matchesSelector) return matchesSelector.call(element, selector);

      // 如果你的环境不支持上面的查询函数，那么你可以通过 querySelectorAll 来查询（也就是下面的 zepto.qsa ）
      var match,
        parent = element.parentNode,
        temp = !parent;
      if (temp) (parent = tempParent).appendChild(element);
      match = ~zepto.qsa(parent, selector).indexOf(element);
      temp && tempParent.removeChild(element);
      return match;
    };

    // 获取变量的数据类型（其实这里我更喜欢 typeof 和 instanceof 当然仅代表个人观点）
    function type(obj) {
      return obj == null
        ? String(obj)
        : class2type[toString.call(obj)] || "object";
    }

    // 判断是不是函数
    function isFunction(value) {
      return type(value) == "function";
    }

    // 判断传递的变量是不是 window 对象
    function isWindow(obj) {
      return obj != null && obj == obj.window;
    }

    // 判断传入的变量是不是 document 对象
    function isDocument(obj) {
      // 其中 DOCUMENT_NODE 代表 9 对应于 节点类型中的 Document（更加详细的节点类型可自行查看 nodeType）
      return obj != null && obj.nodeType == obj.DOCUMENT_NODE;
    }

    // 判断是不是对象
    function isObject(obj) {
      return type(obj) == "object";
    }

    // 因为数组也可以被判定为 object ，这里为了保证传递的是一个真正的 对象，通过一个简单的原型来进行判定
    /**
     * [].__proto__ === Array.prototype   // true
     * ({}).__proto__ === Object.prototype // true
     * [].__proto__ == Object.prototype // false
     */
    function isPlainObject(obj) {
      return (
        isObject(obj) &&
        !isWindow(obj) &&
        // Object.getPrototypeOf 这个函数用于获取参数 obj 的原型
        Object.getPrototypeOf(obj) == Object.prototype
      );
    }

    // 只要满足两个条件就可以判定属于类数组
    /**
     * 1、拥有 length 属性
     * 2、可以遍历
     */
    function likeArray(obj) {
      // 这里的 !! 是把 obj 转变成为 boolean 值
      var length = !!obj && "length" in obj && obj.length,
        type = $.type(obj);

      return (
        "function" != type &&
        !isWindow(obj) &&
        ("array" == type ||
          length === 0 ||
          (typeof length == "number" && length > 0 && length - 1 in obj))
      );
    }

    // 剔出数组中的 null、undefined 元素,注意这里是不包含 "" 的，虽然 "" 和 null 都可以表示 false 但是 "" 和 null 是不等的
    function compact(array) {
      // 这里的 filter 对应于 Array.prototype.filter
      return filter.call(array, function (item) {
        return item != null;
      });
    }

    // 如果是类数组的形式 就变成真数组
    function flatten(array) {
      // $.fn.concat 对应于 Array.prototype.concat
      return array.length > 0 ? $.fn.concat.apply([], array) : array;
    }

    // 将相关字符转变成为驼峰命名方式，比如 font-size => fontSize
    camelize = function (str) {
      return str.replace(/-+(.)?/g, function (match, chr) {
        return chr ? chr.toUpperCase() : "";
      });
    };

    // 将相关字符转变成为短横线命名法  ，比如 fontSize => font-size
    function dasherize(str) {
      return str
        .replace(/::/g, "/")
        .replace(/([A-Z]+)([A-Z][a-z])/g, "$1_$2")
        .replace(/([a-z\d])([A-Z])/g, "$1_$2")
        .replace(/_/g, "-")
        .toLowerCase();
    }

    // 去除数组中的重复项
    uniq = function (array) {
      return filter.call(array, function (item, idx) {
        // indexOf 只要匹配到一次就不会继续寻找
        return array.indexOf(item) == idx;
      });
    };

    // 如果包含这个键 ，那么直接返回，否则将键 作为一个正则返回（可以匹配前后有空格的情况）
    function classRE(name) {
      // classCache 在上面中已经定义过了 classCache = {}
      return name in classCache
        ? classCache[name]
        : // \s 匹配任何不可见字符 可以匹配前后有空格的情况
          (classCache[name] = new RegExp("(^|\\s)" + name + "(\\s|$)"));
    }

    // 判断是否需要添加 px ，在前文中 cssNumber 定义了诸如 font-weight 这类是不需要 px 的
    function maybeAddPx(name, value) {
      // dasherize 这个函数就是上面定义的用于 转变成为短横线命名
      return typeof value == "number" && !cssNumber[dasherize(name)]
        ? value + "px"
        : value;
    }

    // 获取DOM元素 display 的样式属性（以对象的形式返回）
    function defaultDisplay(nodeName) {
      var element, display;
      // elementDisplay 在前面定义过 elementDisplay = {}
      if (!elementDisplay[nodeName]) {
        element = document.createElement(nodeName);
        document.body.appendChild(element);

        // getComputedStyle 函数的作用是获取 DOM 元素的样式
        // getPropertyValue 方法返回指定的 CSS 属性的值
        // 这里只关注了元素的 display 属性
        display = getComputedStyle(element, "").getPropertyValue("display");
        element.parentNode.removeChild(element);
        display == "none" && (display = "block");
        elementDisplay[nodeName] = display;
      }
      return elementDisplay[nodeName];
    }

    // 以数组形式返回子元素
    function children(element) {
      // slice  对应于前面定义的 Array.prototype.slice
      return "children" in element
        ? slice.call(element.children)
        : // 这里的 map 是下文中定义的用于遍历每一个元素进行相应处理，最终是返回一个数组
          $.map(element.childNodes, function (node) {
            if (node.nodeType == 1) return node;
          });
    }

    // 这个 Z 函数的作用？？？
    function Z(dom, selector) {
      var i,
        len = dom ? dom.length : 0;
      for (i = 0; i < len; i++) this[i] = dom[i];
      this.length = len;
      this.selector = selector || "";
    }

    /* 通过html字符串获取文档碎片 */
    zepto.fragment = function (html, name, properties) {
      var dom, nodes, container;

      // 上面定义了 singleTagRE = /^<(\w+)\s*\/?>(?:<\/\1>|)$/ 用于匹配纯标签（不包含属性和值）比如：<img /> 、<span></span> 等
      // 如果匹配到了，创建一个一摸一样的标签出来
      if (singleTagRE.test(html)) dom = $(document.createElement(RegExp.$1));

      if (!dom) {
        // 上面定义的 tagExpanderRE = /<(?!area|br|col|embed|hr|img|input|link|meta|param)(([\w:]+)[^>]*)\/>/gi,用于匹配<input/>、<img/>、<br/>等标签
        // 如果匹配到了单标签 就用正常结束标签替换 比如 <br/> => <br></br>
        if (html.replace) html = html.replace(tagExpanderRE, "<$1></$2>");
        // 上面定义了 fragmentRE = /^\s*<(\w+|!)[^>]*>/  用于取出传入的代码片段中的第一个标签代码
        if (name === undefined) name = fragmentRE.test(html) && RegExp.$1;
        if (!(name in containers)) name = "*";

        container = containers[name];
        container.innerHTML = "" + html;
        dom = $.each(slice.call(container.childNodes), function () {
          container.removeChild(this);
        });
      }

      if (isPlainObject(properties)) {
        nodes = $(dom);
        $.each(properties, function (key, value) {
          /*
          methodAttributes = [
            "val",
            "css",
            "html",
            "text",
            "data",
            "width",
            "height",
            "offset",
          ],
           */
          if (methodAttributes.indexOf(key) > -1) nodes[key](value);
          else nodes.attr(key, value);
        });
      }

      return dom;
    };

    // `$.zepto.Z` swaps out the prototype of the given `dom` array
    // of nodes with `$.fn` and thus supplying all the Zepto functions
    // to the array. This method can be overridden in plugins.
    zepto.Z = function (dom, selector) {
      return new Z(dom, selector);
    };

    // `$.zepto.isZ` should return `true` if the given object is a Zepto
    // collection. This method can be overridden in plugins.
    zepto.isZ = function (object) {
      return object instanceof zepto.Z;
    };

    // `$.zepto.init` is Zepto's counterpart to jQuery's `$.fn.init` and
    // takes a CSS selector and an optional context (and handles various
    // special cases).
    // This method can be overridden in plugins.
    // 将传递进去的选择器进行节点选择以后返回 zepto 对象
    zepto.init = function (selector, context) {
      var dom;
      // If nothing given, return an empty Zepto collection
      if (!selector) return zepto.Z();
      // 如果传递进来的是一个字符串，那么进行相关的元素选择
      else if (typeof selector == "string") {
        selector = selector.trim();
        // If it's a html fragment, create nodes from it
        // Note: In both Chrome 21 and Firefox 15, DOM error 12
        // is thrown if the fragment doesn't begin with <
        if (selector[0] == "<" && fragmentRE.test(selector))
          (dom = zepto.fragment(selector, RegExp.$1, context)),
            (selector = null);
        // If there's a context, create a collection on that context first, and select
        // nodes from there
        else if (context !== undefined) return $(context).find(selector);
        // 如果是一个CSS选择器，就使用这个选择器进行节点选择 querySelectorAll
        else dom = zepto.qsa(document, selector);
      }
      // 如果传递过来的是一个函数，那么当DOM节点准备好以后，就执行它
      else if (isFunction(selector)) return $(document).ready(selector);
      // 如果已经是一个 zepto 对象，那就直接返回它本身
      else if (zepto.isZ(selector)) return selector;
      else {
        // normalize array if an array of nodes is given
        // 这里的compact 是上面定义的用于剔除无效项数据
        if (isArray(selector)) dom = compact(selector);
        // Wrap DOM nodes.
        // 如果是一个对象就把它包裹成为一个数组
        else if (isObject(selector)) (dom = [selector]), (selector = null);
        // If it's a html fragment, create nodes from it
        // 上面定义了 fragmentRE = /^\s*<(\w+|!)[^>]*>/  用于取出传入的代码片段中的第一个标签代码
        else if (fragmentRE.test(selector))
          (dom = zepto.fragment(selector.trim(), RegExp.$1, context)),
            (selector = null);
        // If there's a context, create a collection on that context first, and select
        // nodes from there
        else if (context !== undefined) return $(context).find(selector);
        // And last but no least, if it's a CSS selector, use it to select nodes.
        else dom = zepto.qsa(document, selector);
      }
      // create a new Zepto collection from the nodes found
      // 从查询出来的节点中创建 zepto 对象
      return zepto.Z(dom, selector);
    };

    // `$` will be the base `Zepto` object. When calling this
    // function just call `$.zepto.init, which makes the implementation
    // details of selecting nodes and creating Zepto collections
    // patchable in plugins.
    $ = function (selector, context) {
      return zepto.init(selector, context);
    };

    // 深拷贝
    function extend(target, source, deep) {
      for (key in source)
        if (deep && (isPlainObject(source[key]) || isArray(source[key]))) {
          if (isPlainObject(source[key]) && !isPlainObject(target[key]))
            target[key] = {};
          if (isArray(source[key]) && !isArray(target[key])) target[key] = [];
          extend(target[key], source[key], deep);
        } else if (source[key] !== undefined) target[key] = source[key];
    }

    // Copy all but undefined properties from one or more
    // objects to the `target` object.

    // 将多个 source 的属性装载到 target 中
    // 一般传入的参数会是：
    // (targetObj, srcObj1, srcObj2, srcObj1...)
    $.extend = function (target) {
      var deep,
        args = slice.call(arguments, 1);
      if (typeof target == "boolean") {
        deep = target;
        target = args.shift();
      }
      args.forEach(function (arg) {
        extend(target, arg, deep);
      });
      return target;
    };

    // `$.zepto.qsa` is Zepto's CSS selector implementation which
    // uses `document.querySelectorAll` and optimizes for some special cases, like `#id`.
    // This method can be overridden in plugins.

    // 根据传递的元素类型 和 选择器进行元素选择
    zepto.qsa = function (element, selector) {
      var found,
        maybeID = selector[0] == "#", // 判定是否是 id 选择器
        maybeClass = !maybeID && selector[0] == ".", // 判定是否是 class 选择器
        // 只取出名称部分
        nameOnly = maybeID || maybeClass ? selector.slice(1) : selector, // Ensure that a 1 char tag name still gets checked
        // simpleSelectorRE = /^[\w-]*$/,  匹配一个包括（字母、数组、下划线、-）的字符串
        isSimple = simpleSelectorRE.test(nameOnly);

      // 先判定是否选择器是 id 选择器，如果是那么通过 getElementById 选取元素然后包裹成为数组
      return element.getElementById && isSimple && maybeID // Safari DocumentFragment doesn't have getElementById
        ? (found = element.getElementById(nameOnly))
          ? [found]
          : []
        : //  如果传递的不是 id 选择器 那么判定元素的类型是否是 元素节点 或 DOM树根节点 或 文档片段
        // 如果 不是 那么就直接返回空数组
        element.nodeType !== 1 &&
          element.nodeType !== 9 &&
          element.nodeType !== 11
        ? []
        : // 如果是元素节点或DOM根节点或文档片段，那么排除上面的 id 选择器，就可以尝试选用 class 选择器，或标签选择器或  querySelectorAll
          slice.call(
            isSimple && !maybeID && element.getElementsByClassName // DocumentFragment doesn't have getElementsByClassName/TagName
              ? maybeClass
                ? element.getElementsByClassName(nameOnly) // If it's simple, it could be a class
                : element.getElementsByTagName(selector) // Or a tag
              : element.querySelectorAll(selector) // Or it's not simple, and we need to query all
          );
    };

    // 根据selector 筛选nodes
    function filtered(nodes, selector) {
      return selector == null ? $(nodes) : $(nodes).filter(selector);
    }

    // 判定传递的祖先元素中是否包含相应子孙节点
    // 首先判定浏览器是否支持 contains 方法
    $.contains = document.documentElement.contains
      ? function (parent, node) {
          // 如果支持那么就返回判断的结果
          return parent !== node && parent.contains(node);
        }
      : function (parent, node) {
          // 如果不支持就循环一步一步往上寻找父节点，判断每次迭代的节点是否就是需要比较的父节点
          while (node && (node = node.parentNode))
            if (node === parent) return true;
          return false;
        };

    // 用于改变函数的执行环境
    function funcArg(context, arg, idx, payload) {
      return isFunction(arg) ? arg.call(context, idx, payload) : arg;

    //既可以设置属性也可以删除属性，一个函数就搞定
    function setAttribute(node, name, value) {
      value == null
        ? node.removeAttribute(name)
        : node.setAttribute(name, value);

    // access className property while respecting SVGAnimatedString
    // 返回节点的类名
    function className(node, value) {
      var klass = node.className || "",
        svg = klass && klass.baseVal !== undefined;

      if (value === undefined) return svg ? klass.baseVal : klass;
      svg ? (klass.baseVal = value) : (node.className = value);
    }

    // "true"  => true
    // "false" => false
    // "null"  => null
    // "42"    => 42
    // "42.5"  => 42.5
    // "08"    => "08"
    // JSON    => parse if valid
    // String  => self

    // 源码中的注释已经很清楚了
    function deserializeValue(value) {
      try {
        return value
          ? value == "true" ||
              (value == "false"
                ? false
                : value == "null"
                ? null
                : +value + "" == value
                ? +value
                : /^[\[\{]/.test(value)
                ? $.parseJSON(value)
                : value)
          : value;
      } catch (e) {
        return value;
      }
    }

    $.isNumeric = function (val) {

      return (

          type != "boolean" &&
          (type != "string" || val.length) &&
          !isNaN(num) &&

      return emptyArray.indexOf.call(array, elem, i);

    $.camelCase = camelize;

    $.trim = function (str) {
      return str == null ? "" : String.prototype.trim.call(str);
    };

    // plugin compatibility

    $.uuid = 0;
    $.support = {};
    $.expr = {};
    $.noop = function () {};

    // 将传递的元素都经过callback 的洗礼
    $.map = function (elements, callback) {
      var value,
        values = [],
        i,
        key;
      // 类数组 或者 数组
      if (likeArray(elements))
        for (i = 0; i < elements.length; i++) {
          // 用传递过来的 callback 对元素进行操作，满足条件的元素会被返回
          value = callback(elements[i], i);
          if (value != null) values.push(value);
        }
      else
        for (key in elements) {
          value = callback(elements[key], key);
          if (value != null) values.push(value);
        }

      // 最后无论 values 是类数组还是数组最后都要返回数组
      return flatten(values);
    };

    // 遍历每一个存在于 elements 中的元素都执行一遍 callback，最终还是返回 elements
    $.each = function (elements, callback) {
      var i, key;
      if (likeArray(elements)) {
        for (i = 0; i < elements.length; i++)
          if (callback.call(elements[i], i, elements[i]) === false)
            return elements;
      } else {
        // 对于数组和对象都可以使用 for ... in
        for (key in elements)
          if (callback.call(elements[key], key, elements[key]) === false)
            return elements;
      }

      return elements;
    };

    // 使用callback过滤数组，
    $.grep = function (elements, callback) {
      return filter.call(elements, callback);
    };

    if (window.JSON) $.parseJSON = JSON.parse;

    // Populate the class2type map

    $.each(
      "Boolean Number String Function Array Date RegExp Object Error".split(
        " "
      ),
      // 主要用于格式化输出 对象的类型
      /* 比如
      {
        '[object number]': 'number',
        '[object string]': 'string',
        ...
      } */
      function (i, name) {
        // 上文中定义了 class2type 是 {}
        class2type["[object " + name + "]"] = name.toLowerCase();
      }
    );

    // Define methods that will be available on all
    // Zepto collections

    // 主要是定义各种数组的内置方法。因为一个zepto对象本身是一个类数组
    $.fn = {
      constructor: zepto.Z,
      length: 0,

      // Because a collection acts like an array
      // copy over these useful array functions.
      forEach: emptyArray.forEach,
      reduce: emptyArray.reduce,
      push: emptyArray.push,
      sort: emptyArray.sort,
      splice: emptyArray.splice,
      indexOf: emptyArray.indexOf,
      concat: function () {
        var i,
          value,
          args = [];
        for (i = 0; i < arguments.length; i++) {
          value = arguments[i];
          args[i] = zepto.isZ(value) ? value.toArray() : value;
        }
        return concat.apply(zepto.isZ(this) ? this.toArray() : this, args);
      },

                $(selector);

      if (0 in arguments) this.bind("submit", callback);
      else if (this.length) {
        var event = $.Event("submit");
        this.eq(0).trigger(event);
        if (!event.isDefaultPrevented()) this.get(0).submit();
      }
      return this;
    };
